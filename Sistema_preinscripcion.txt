CREATE DATABASE ColegioInscripciones;
USE ColegioInscripciones;

CREATE TABLE grados (
    id_grado INT PRIMARY KEY AUTO_INCREMENT,
    nombre_grado VARCHAR(50) NOT NULL,
    cupos_disponibles INT
);

CREATE TABLE ubicacion (
    id_ubicacion INT PRIMARY KEY AUTO_INCREMENT,
    municipio VARCHAR(50),
    calle VARCHAR(50),
    avenida VARCHAR(50),
    num_casa VARCHAR(50)
);

CREATE TABLE representantes (
    id_representante INT PRIMARY KEY AUTO_INCREMENT,
    cedula VARCHAR(20) UNIQUE NOT NULL,
    nombre_completo VARCHAR(150) NOT NULL,
    parentesco VARCHAR(50),
    telefono_principal VARCHAR(20) NOT NULL,
    correo VARCHAR(100),
    id_ubicacion INT,
    FOREIGN KEY (id_ubicacion) REFERENCES ubicacion(id_ubicacion)
);

CREATE TABLE estudiantes (
    id_estudiante INT PRIMARY KEY AUTO_INCREMENT,
    id_representante INT,
    nombres VARCHAR(100) NOT NULL,
    apellidos VARCHAR(100) NOT NULL,
    fecha_nacimiento VARCHAR(50),
    alergias_condiciones TEXT,
    FOREIGN KEY (id_representante) REFERENCES Representantes(id_representante)
);

CREATE TABLE cargo (
    id_cargo INT PRIMARY KEY AUTO_INCREMENT,
    nombre_cargo VARCHAR(50)
);

CREATE TABLE docente (
    id_docente INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(50),
    cedula VARCHAR(50),
    titulo VARCHAR(50),
    id_cargo INT,
    FOREIGN KEY (id_cargo) REFERENCES cargo(id_cargo)
);

CREATE TABLE preinscripciones (
    id_preinscripcion INT PRIMARY KEY AUTO_INCREMENT,
    id_estudiante INT,
    id_grado_solicitado INT,
    fecha_solicitud VARCHAR(50),
    estatus VARCHAR(50),
    promedio_anterior DECIMAL(4,2),
    FOREIGN KEY (id_estudiante) REFERENCES Estudiantes(id_estudiante),
    FOREIGN KEY (id_grado_solicitado) REFERENCES Grados(id_grado)
);

CREATE TABLE r_r_p (
    id_r_r_p INT PRIMARY KEY AUTO_INCREMENT,
    id_representante INT,
    id_preinscripcion INT,
    FOREIGN KEY (id_representante) REFERENCES representantes(id_representante),
    FOREIGN KEY (id_preinscripcion) REFERENCES preinscripciones(id_preinscripcion)
);

CREATE TABLE entrevistas (
    id_entrevista INT PRIMARY KEY AUTO_INCREMENT,
    fecha VARCHAR(50),
    entrevistador VARCHAR(100),
    id_docente INT,
    id_representante_inscripcion INT,
    resultado VARCHAR(50),
    FOREIGN KEY (id_docente) REFERENCES docente(id_docente),
    FOREIGN KEY (id_representante_inscripcion) REFERENCES r_r_p(id_r_r_p)
);

CREATE TABLE direccion (
    id_direccion INT AUTO_INCREMENT PRIMARY KEY,
    calle VARCHAR(50),
    avenida VARCHAR(50)
);

CREATE TABLE usuarios (
    id_usuarios INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(50),
    id_direccion INT,
    FOREIGN KEY (id_direccion) REFERENCES direccion(id_direccion)
);

CREATE TABLE rol (
    id_rol INT AUTO_INCREMENT PRIMARY KEY,
    cargo VARCHAR(50)
);

CREATE TABLE r_u_r (
    id_r_u_r INT AUTO_INCREMENT PRIMARY KEY,
    id_usuarios INT,
    id_rol INT,
    FOREIGN KEY (id_usuarios) REFERENCES usuarios(id_usuarios),
    FOREIGN KEY (id_rol) REFERENCES rol(id_rol)
);


CREATE TABLE modulo (
    id_modulo INT AUTO_INCREMENT PRIMARY KEY,
    nombre_modulo VARCHAR(50),
    codigo_interno VARCHAR(50),
    orden INT
);

CREATE TABLE paginas (
    id_paginas INT AUTO_INCREMENT PRIMARY KEY,
    url VARCHAR(50),
    nombre_pagina VARCHAR(50)
);

CREATE TABLE r_m_p (
    id_r_m_p INT AUTO_INCREMENT PRIMARY KEY,
    id_usuario_rol INT,
    id_modulo INT,
    id_paginas INT,
    FOREIGN KEY (id_modulo) REFERENCES modulo(id_modulo),
    FOREIGN KEY (id_paginas) REFERENCES paginas(id_paginas),
    FOREIGN KEY (id_usuario_rol) REFERENCES r_u_r(id_r_u_r)
);

CREATE TABLE permisos (
    id_permisos INT AUTO_INCREMENT PRIMARY KEY,
    accion VARCHAR(50)
);

CREATE TABLE r_mp_per (
    id_r_mp_per INT AUTO_INCREMENT PRIMARY KEY,
    id_permisos INT,
    id_modulo_pagina INT,
    FOREIGN KEY (id_permisos) REFERENCES permisos(id_permisos),
    FOREIGN KEY (id_modulo_pagina) REFERENCES r_m_p(id_r_m_p)
);
